20차시: 단어 빈도 분석

예제

text = '''
Python is fun and powerful.
Python lets you write clear and concise code.
Fun with Python!
'''
for ch in ",.!?":
    text = text.replace(ch, " ")
words = [w for w in text.lower().split() if w]
freq = {}
for w in words:
    freq[w] = freq.get(w, 0) + 1
items = sorted(freq.items(), key=lambda x: (-x[1], x[0]))
print("단어 개수:", len(words))
for w, c in items:
    print(f"{w}: {c}")
