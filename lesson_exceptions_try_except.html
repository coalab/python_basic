13차시: 예외 처리

예제

def safe_divide(a, b):
    try:
        return a / b
    except ZeroDivisionError:
        print("0으로는 나눌 수 없어요!")
        return None
print(safe_divide(10, 2))
print(safe_divide(5, 0))
