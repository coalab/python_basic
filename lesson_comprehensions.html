16차시: 컴프리헨션

예제

nums = list(range(1, 11))
squares = [n*n for n in nums]
evens = [n for n in nums if n % 2 == 0]
words = ["apple","banana","avocado","grape"]
a_starts = [w for w in words if w.startswith("a")]
scores = {"kim":88,"lee":91,"park":73}
passed = {name:sc for name,sc in scores.items() if sc>=80}
print(squares, evens, a_starts, passed)
